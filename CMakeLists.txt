CMAKE_MINIMUM_REQUIRED(VERSION 3.0)

# Set version
IF(NOT DEFINED CYBERTRON_VERSION_MAJOR)
    SET(CYBERTRON_VERSION_MAJOR 0)
ENDIF()
IF(NOT DEFINED CYBERTRON_VERSION_MINOR)
    SET(CYBERTRON_VERSION_MINOR 1)
ENDIF()
IF(NOT DEFINED CYBERTRON_VERSION_PATCH)
    SET(CYBERTRON_VERSION_PATCH 0)
ENDIF()
IF(NOT DEFINED CYBERTRON_VERSION_BUILD)
    SET(CYBERTRON_VERSION_BUILD 0)
ENDIF()
IF(NOT PACKAGE_VERSION)
    SET(PACKAGE_VERSION "${DF_VERSION_MAJOR}.${DF_VERSION_MINOR}.${DF_VERSION_PATCH}.${DF_VERSION_BUILD}")
ENDIF
